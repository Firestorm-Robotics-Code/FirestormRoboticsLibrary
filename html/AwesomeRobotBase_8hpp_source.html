166 static void* thread_function(void* _robot){
167 AwesomeRobot* robot = (AwesomeRobot*)_robot;
168 pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL); // Set how the thread should handle being cancelled
169 // The Asynchronous cancel type means it can be canceled at any time; the alternative "deferred" means it waits until a library function checks if there's a queued cancel request.
170 // We want to be able to run infinite loops in here, so asynchronous is the best and only option.
171 double lastTime = (double)frc::Timer::GetFPGATimestamp();
172 float frameSecs = 1/robot -> activeMode -> hz; // Maintain this many seconds per frame
173 while (true){ // Frequency loop
174 // note that it's totally safe to run infinite loops, use delays, and etc in your Thread() code! This is just a frequency controller.
175 robot -> activeMode -> Thread();
176 double timeElapsed = (double)frc::Timer::GetFPGATimestamp() - lastTime; // Seconds the frame took
177 if (timeElapsed < frameSecs){ // If it took too long, don't even bother waiting
178 sleep((frameSecs - timeElapsed)); // Wait to fill in the missed 1/hz seconds.
179 }
180 }
181 }

187 void switchMode(RobotMode* toMode){
188 if (activeMode){ // If the pointer is actually set and isn't initial 0
189 pthread_cancel(thread); // Only ever called if ActiveMode is non-NULL; e.g. switchMode has already been called and a thread is actually running
190 pthread_join(thread, NULL); // It isn't guaranteed that cancellation will be immediate, even though the thread cancel type is asynchronous
191 // It is, however, guaranteed that the thread will be exited as soon as pthread_join returns
192 // NULL means don't bother with the return value of the pthread's function
193 activeMode -> End(); // End() may do cleanup work concerning threads (there are situations where this is acceptable, such as when memory has been allocated by Start specifically for the usage of that thread),
194 // so it's important that the thread really be exited by the time End() runs.
195 }
196 activeMode = toMode;
197 toMode -> Start();
198 modeThread.InDisabled(false);
199 modeThread.InAutonomous(false);
200 modeThread.InTeleop(false);
201 modeThread.InTest(false);
202 if (activeMode == disabled){
203 modeThread.InDisabled(true);
204 }
205 else if (activeMode == teleop){
206 modeThread.InTeleop(true);
207 }
208 else if (activeMode == autonomous){
209 modeThread.InAutonomous(true);
210 }
211 else if (activeMode == test){
212 modeThread.InTest(true);
213 }
214 // Start the thread for that mode
215 if (pthread_create(&thread, NULL, thread_function, this) != 0){
216 activeMode -> noThread = true;
217 std::cout << "Thread creation failed! Nothread flag set in the active robot mode." << std::endl;
218 }

220 activeMode -> noThread = false;
221 }
