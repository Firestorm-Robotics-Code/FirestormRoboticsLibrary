135 double DoMath(double FE){
136 double error = getError(setPoint, curPos);
137
138 double p = error * constants.P; // This does not need to be adjusted for FE
139
140 if (fabs(error) <= constants.iZone || constants.iZone == 0){ // no clue, I'm basically copy pasting. looks like IZone is a "zone" in which the I coefficient applies.
141 iState += (error * constants.I) * FE; // *FE means that, if error * constants.I is 2, it will only actually gain 2 after 1 second/hz is passed. (20 ms by default). This keeps it smooth.
142 // This kind of thing is used all throughout platformer; very tested and stable
143 }
144 else{
145 iState = 0;
146 }
147
148 double d = (error - previousError); // Not really sure how this can be related to FE; I suppose I'll just leave it how it is and work it out later.
149 previousError = error;
150 d *= constants.D;
151
152 float f = setPoint * constants.F;
153
154 return p + iState + d + f;
155 }
